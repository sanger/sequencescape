def update_request
  if self.request?
    request = self.eventful
    unless request.nil? or request.failed? or request.cancelled?
      if self.family == "fail"
        request.fail!
      elsif self.family == "pass" # && !request.project.nil?
        request.pass!
      end
    end
  end
end
