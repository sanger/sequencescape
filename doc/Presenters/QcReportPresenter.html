def csv_headers
  @csv << [REPORT_IDENTITY, VERSION]
  @csv << [I18n.t('qc_reports.fixed_content')]
  @csv << [I18n.t('qc_reports.instruction')]
  each_header do |pair|
    @csv << pair
  end
end
