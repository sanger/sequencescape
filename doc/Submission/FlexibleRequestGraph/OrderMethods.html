def build_request_graph!(multiplexing_assets = nil, &block)
  ActiveRecord::Base.transaction do
    RequestChain.new(self, assets, multiplexing_assets).tap do |chain|
      chain.build!
      yield chain.multiplexing_assets if chain.multiplexed?
    end
  end
end
