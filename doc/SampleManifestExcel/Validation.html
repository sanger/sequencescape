def update(attributes = {})
  if range_required?
    options[:formula1] = attributes[:range].absolute_reference
  end

  if attributes[:worksheet].present?
    @worksheet_validation = attributes[:worksheet].add_data_validation(attributes[:reference], options)
  end

end
